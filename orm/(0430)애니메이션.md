      alignment: FractionalOffset.center,
      child: GestureDetector(
        onPanUpdate: (details) {
          setState(() {
            // y = y - details.delta.dx / 100;

            // x 축을 따라 세로로 스와이프한다.
            x = x + details.delta.dy / 100;
          });
        },
        child: Container(
          color: Colors.red,
          height: 200.0,
          width: 200.0,
        ),
      ),
    );
  }
